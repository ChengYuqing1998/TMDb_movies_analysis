def splitlist(list):
    alist=[]
    a=0
    for sublist in list:
        try:
            for i in sublist:
                alist.append(i)
        except TypeError:
            alist.append(sublist)
    for i in alist:
        if type(i)==type([]):
            a=+1
            break
    if a==1:
        return splitlist(alist)
    if a==0:
        return alist
